99template<class LockGuard>
100bool Timer<LockGuard>::processNextTimeout(uint32_t const now)
101{
102 Timeout* timeout = nullptr;
103 int32_t diffTimeout = 0;
104 {
105 LockGuard const scopedLock;
106 if (!_timeoutList.empty())
107 {
108 diffTimeout = diff(_timeoutList.front()._time, now);
109 if (diffTimeout <= 0)
110 {
111 timeout = &_timeoutList.front();
112 _timeoutList.pop_front();
113 }
114 else
115 {
116 return false;
117 }
118 }
119 else
120 {
121 return false;
122 }
123 }
124
125 rescheduleCyclicTimeout(*timeout, now);
126 timeout->expired();
127 return diffTimeout == 0U;
128}

190template<class LockGuard>
191bool Timer<LockGuard>::addTimeout(
192 Timeout& timeout, uint32_t const absoluteTimeout, uint32_t const cycleTime, uint32_t const now)
193{
194 timeout._time = absoluteTimeout;
195 timeout._cycleTime = cycleTime;
196
197 LockGuard const lock;
198
199 int32_t const timeoutDiff = diff(timeout._time, now);
200 TimeoutList::iterator prev = _timeoutList.before_begin();
201
202 for (TimeoutList::iterator current = _timeoutList.begin(); current != _timeoutList.end();
203 ++current)
204 {
205 int32_t const nextDiff = diff(current->_time, now);
206 if (nextDiff > timeoutDiff)
207 {
208 break;
209 }
210 prev = current;
211 }
212
213 TimeoutList::iterator const insertPos = _timeoutList.insert_after(prev, timeout);
214 return _timeoutList.begin() == insertPos; // will expire before all other
215}
