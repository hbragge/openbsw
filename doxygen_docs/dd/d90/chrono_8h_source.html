692template<typename ToDur, typename Rep, typename Period>
693ToDur duration_cast(duration<Rep, Period> const& d)
694{
695 using todur_t = typename enable_if<internal::_is_duration<ToDur>::value, ToDur>::type;
696 using to_period_t = typename ToDur::period;
697 using to_rep_t = typename ToDur::rep;
698 using cf_t = ::estd::ratio_divide<Period, to_period_t>;
699 using cr_t = typename ::estd::internal::_common_integral_type<
700 typename ::estd::internal::_common_integral_type<to_rep_t, Rep>::type,
701 intmax_t>::type;
702 using dc_t = internal::_duration_cast_impl<todur_t, cf_t, cr_t>;
703 return dc_t::_cast(d);
704}
