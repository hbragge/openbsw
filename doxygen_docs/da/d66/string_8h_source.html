32  * Convert a string as a slice to an integer of type T, where T must be a primitive number type.
33  * Slices must not have a null-terminating character.
34  */
35 template<typename T>
36 ::estd::result<T, ConvertError> strtonum(::estd::slice<uint8_t const> const data, uint8_t base = 10)
37 {
38  (void)guaranteePrimitive<T>();
39 
40  uint64_t value = 0;
41  size_t index = 0;
42  bool const negation = (std::is_signed<T>::value && (data[0] == static_cast<uint8_t>('-')));
43  if (negation || (data[0] == static_cast<uint8_t>('+')))
44  {
45  index++;
46  }
47 
48  // Alphabetic mask of ASCII is & 0xDF (11011111), even after negating '0'
49  constexpr uint8_t alphaMask = static_cast<uint8_t>(223);
50  if ((base == 16) || (base == 0))
51  {
52  if ((index + 2U < data.size())
53  && ((data[index + 1U] & alphaMask) == static_cast<uint8_t>('X')))
54  {
55  index += 2;
56  base = 16;
57  }
58  else if (base == 0)
59  {
60  base = 10;
61  }
62  else
63  {
64  // Base == 16 but not 0x/0X
65  }
66  }
67  else if ((base < 2) || (base > 36))
68  {
69  return ConvertError::InvalidBase;
70  }
71  else
72  {
73  // Base is set.
74  }
75 
76  if (((data.size()) == 0U) || (data.size() - index == 0U))
77  {
78  // Token does not include a number
79  return ConvertError::InvalidToken;
80  }
81  // Clear leading zeros, if loops til data.size() == index, returns 0 (as it was all zeros)
82  while ((index < data.size()) && (data[index] == static_cast<uint8_t>('0')))
83  {
84  index++;
85  }
86  uint64_t const maxValue = negation ? ::estd::abs<uint64_t>(std::numeric_limits<T>::min())
87  : static_cast<uint64_t>(std::numeric_limits<T>::max());
88  // Multiplication limit for any base is the integer division of the max value and base
89  uint64_t const mulLimit = maxValue / static_cast<uint64_t>(base);
90  // For base X, the limit for adding is in the range 0:X-1
91  uint8_t addLimit = base - 1;
92  constexpr uint8_t zeroChar = static_cast<uint8_t>('0');
93  constexpr uint8_t aChar = static_cast<uint8_t>('A');
94  for (; index < data.size(); index++)
95  {
96  if (value >= mulLimit)
97  {
98  if (value != mulLimit)
99  {
100  return ConvertError::Overflow;
101  }
102  // If Value == mulLimit, the last add limit is the remainder of the integer division
103  // done while calculating the multiplication limit.
104  addLimit = static_cast<uint8_t>(maxValue % base);
105  }
106  uint8_t character = data[index] - zeroChar;
107 
108  if (character > 9)
109  {
110  character = (character & alphaMask) - (aChar - zeroChar - 10);
111  if (character < 10)
112  {
113  return ConvertError::InvalidToken;
114  }
115  }
116 
117  if (character > addLimit)
118  {
119  // If the addLimit has changed in the last mulLimit check, this could be an overflow, so
120  // check if the token is within the allowed chars of the base. Otherwise an invalid
121  // token.
122  return (character >= (base)) ? ConvertError::InvalidToken : ConvertError::Overflow;
123  }
124  value = (value * base) + character;
125  }
126 
127  T const returnVal = ((negation) ? static_cast<T>(-value) : static_cast<T>(value));
128  // If signed type, but positive, the cast of abs(minVal) results in a negative number, and thus
129  // overflow as abs(minVal) == maxValue + 1.
130  if (negation == (returnVal >= 0))
131  {
132  return ConvertError::Overflow;
133  }
134  return returnVal;
135 }
