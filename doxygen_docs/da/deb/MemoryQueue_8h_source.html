41template<size_t CAPACITY, size_t MAX_ELEMENT_SIZE, typename SIZE_TYPE = uint16_t>
42class MemoryQueue
43{
44 static_assert(CAPACITY >= MAX_ELEMENT_SIZE + sizeof(SIZE_TYPE), "");
45
46 struct RxData
47 {
48 std::atomic<size_t> received{0U};
49 };
50
51 RxData rx;
52
53 struct TxData
54 {
55 std::atomic<size_t> sent{0U};
56 ::estd::array<uint8_t, CAPACITY> data;
57 size_t allocated{0U};
58 size_t minAvailable{CAPACITY};
59
60 size_t available(RxData const& rxData) const;
61 };
62
63 TxData tx;
64
65 static size_t advanceIndex(size_t index, size_t const size)
66 {
67 index = (index + size + sizeof(SIZE_TYPE)) % (2 * CAPACITY);
68
69 // Check if enough contiguous space is available at the end of the array and wrap
70 // otherwise skipping the too small piece of memory.
71 size_t const space = (((2 * CAPACITY) - index) % CAPACITY);
72 if (space < (MAX_ELEMENT_SIZE + sizeof(SIZE_TYPE)))
73 {
74 index = (index + space) % (2 * CAPACITY);
75 }
76 return index;
77 }

232template<size_t CAPACITY, size_t MAX_ELEMENT_SIZE, typename SIZE_TYPE>
233::estd::slice<uint8_t>
234MemoryQueue<CAPACITY, MAX_ELEMENT_SIZE, SIZE_TYPE>::Writer::allocate(size_t const size) const
235{
236 // Set allocated zero prevent a subsequent call to commit() to have
237 // effects in case this allocation fails. That is important if this is a
238 // reallocation, i.e. a previous call to allocate() succeeded.
239 _txData.allocated = 0U;
240 if ((size > MAX_ELEMENT_SIZE) || (size == 0))
241 {
242 return {};
243 }
244 auto const freeBytes = available();
245 _txData.minAvailable = ::estd::min(freeBytes, _txData.minAvailable);
246 if (freeBytes == 0)
247 {
248 return {};
249 }
250 size_t const index = _txData.sent.load() % CAPACITY;
251 _txData.allocated = size;
252 return ::estd::memory::as_bytes(&_txData.data[index + sizeof(SIZE_TYPE)], size);
253}

255template<size_t CAPACITY, size_t MAX_ELEMENT_SIZE, typename SIZE_TYPE>
256void MemoryQueue<CAPACITY, MAX_ELEMENT_SIZE, SIZE_TYPE>::Writer::commit()
257{
258 // Prevent accidentally committing random data if allocate has not been called or
259 // previous allocation was unsuccessful.
260 if (_txData.allocated == 0U)
261 {
262 return;
263 }
264 size_t writeIndex = _txData.sent.load();
265 size_t const index = writeIndex % CAPACITY;
266 SIZE_TYPE const size = static_cast<SIZE_TYPE>(_txData.allocated);
267 ::estd::write_be<SIZE_TYPE>(&_txData.data[index], static_cast<SIZE_TYPE>(size));
268
269 writeIndex = advanceIndex(writeIndex, size);
270 _txData.allocated = 0U;
271 // Store writeIndex last to ensure data consistency.
272 _txData.sent.store(writeIndex);
273}

331template<size_t CAPACITY, size_t MAX_ELEMENT_SIZE, typename SIZE_TYPE>
332void MemoryQueue<CAPACITY, MAX_ELEMENT_SIZE, SIZE_TYPE>::Reader::release() const
333{
334 if (empty())
335 {
336 return;
337 }
338 size_t readIndex = _rxData.received.load();
339 size_t const index = readIndex % CAPACITY;
340 SIZE_TYPE const size = ::estd::read_be<SIZE_TYPE>(&_txData.data[index]);
341 readIndex = advanceIndex(readIndex, size);
342 // Store readIndex last to ensure data consistency.
343 _rxData.received.store(readIndex);
344}

367template<size_t CAPACITY, size_t MAX_ELEMENT_SIZE, typename SIZE_TYPE>
368size_t
369MemoryQueue<CAPACITY, MAX_ELEMENT_SIZE, SIZE_TYPE>::TxData::available(RxData const& rxData) const
370{
371 size_t const writeIndex = sent.load();
372 size_t const readIndex = rxData.received.load();
373
374 size_t usedBytes;
375 if (writeIndex < readIndex)
376 {
377 usedBytes = (writeIndex + (2 * CAPACITY)) - readIndex;
378 }
379 else
380 {
381 usedBytes = writeIndex - readIndex;
382 }
383
384 size_t freeBytes = CAPACITY - usedBytes;
385 if (freeBytes < (MAX_ELEMENT_SIZE + sizeof(SIZE_TYPE)))
386 {
387 freeBytes = 0;
388 }
389 return freeBytes;
390}
