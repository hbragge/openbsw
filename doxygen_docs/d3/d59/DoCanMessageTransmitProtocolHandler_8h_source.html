310template<typename FrameIndexType>
311inline TransmitResult DoCanMessageTransmitProtocolHandler<FrameIndexType>::frameSending()
312{
313 if (_state == TransmitState::SEND)
314 {
315 FrameIndexType const nextFrameIndex = _frameIndex + 1U;
316 if ((nextFrameIndex == _blockEnd) && (nextFrameIndex != _frameCount))
317 {
318 _flowControlWaitCount = 0U;
319 return setState(
320 TransmitState::WAIT, FlowControl::EXPECTED, TransmitTimeout::TX_CALLBACK);
321 }
322 return setState(TransmitState::WAIT, FlowControl::UNEXPECTED, TransmitTimeout::TX_CALLBACK);
323 }
324 return setFailed(TransmitMessage::ILLEGAL_STATE, static_cast<uint8_t>(_state));
325}

327template<typename FrameIndexType>
328inline TransmitResult
329DoCanMessageTransmitProtocolHandler<FrameIndexType>::framesSent(FrameIndexType const frameCount)
330{
331 if ((_state != TransmitState::WAIT) || (_timeout != TransmitTimeout::TX_CALLBACK))
332 {
333 return TransmitResult(true).setMessage(
334 TransmitMessage::ILLEGAL_STATE, static_cast<uint8_t>(_state));
335 }
336
337 _frameIndex += frameCount;
338 if (_frameIndex >= _frameCount)
339 {
340 return setState(TransmitState::SUCCESS);
341 }
342
343 if ((frameCount > 1U) && (_frameIndex >= _blockEnd))
344 {
345 _frameIndex = _blockEnd;
346 _flowControl = FlowControl::EXPECTED;
347 }
348 switch (_flowControl)
349 {
350 case FlowControl::RECEIVED_CTS:
351 {
352 return setSend();
353 }
354 case FlowControl::RECEIVED_WAIT:
355 case FlowControl::EXPECTED:
356 {
357 return setState(TransmitState::WAIT, _flowControl, TransmitTimeout::FLOW_CONTROL);
358 }
359 default:
360 {
361 if (_hasMinSeparationTime)
362 {
363 return setState(
364 TransmitState::WAIT, FlowControl::UNEXPECTED, TransmitTimeout::SEPARATION_TIME);
365 }
366 return setSend();
367 }
368 }
369}

371template<typename FrameIndexType>
372inline TransmitResult DoCanMessageTransmitProtocolHandler<FrameIndexType>::handleFlowControl(
373 FlowStatus const flowStatus,
374 uint8_t const blockSize,
375 bool const hasMinSeparationTime,
376 uint8_t const maxWaitCount)
377{
378 if (_flowControl != FlowControl::EXPECTED)
379 {
380 return TransmitResult(false);
381 }
382
383 switch (flowStatus)
384 {
385 case FlowStatus::CTS:
386 {
387 _blockEnd = (blockSize > 0U) ? (_blockEnd + blockSize) : _frameCount;
388 _hasMinSeparationTime = hasMinSeparationTime;
389 if (_timeout == TransmitTimeout::FLOW_CONTROL)
390 {
391 (void)setSend(); // Result of setSend doesn't seem to matter, we know what to return
392 return TransmitResult(true).setActionSet(
393 TransmitActionSetType().set(TransmitAction::STORE_SEPARATION_TIME));
394 }
395
396 _flowControl = FlowControl::RECEIVED_CTS;
397 return TransmitResult(false).setActionSet(
398 TransmitActionSetType().set(TransmitAction::STORE_SEPARATION_TIME));
399 }
400 case FlowStatus::WAIT:
401 {
402 if (_flowControlWaitCount < maxWaitCount)
403 {
404 ++_flowControlWaitCount;
405 if (_timeout == TransmitTimeout::FLOW_CONTROL)
406 {
407 return setState(
408 TransmitState::WAIT, FlowControl::EXPECTED, TransmitTimeout::FLOW_CONTROL);
409 }
410
411 _flowControl = FlowControl::RECEIVED_WAIT;
412 return TransmitResult(false);
413 }
414
415 return setFailed(TransmitMessage::FLOW_CONTROL_WAIT_COUNT_EXCEEDED);
416 }
417 case FlowStatus::OVFLW:
418 {
419 return setFailed(TransmitMessage::FLOW_CONTROL_OVERFLOW);
420 }
421 default:
422 {
423 return setFailed(TransmitMessage::FLOW_CONTROL_INVALID);
424 }
425 }
