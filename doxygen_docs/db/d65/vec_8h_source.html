580template<class T, size_t N>
581template<class InputIterator>
582inline typename std::enable_if<std::is_integral<InputIterator>::value == false, void>::type
583vec<T, N>::assign(InputIterator first, InputIterator const last)
584{
585 clear();
586 for (; (!full()) && (first != last); ++first)
587 {
588 // We don't call emplace_back here to avoid the extra estd_assert per call.
589 // We can dereference first because it's an iterator and not necessarily a pointer. Also,
590 // the previous check != last makes it safe to assume that first can be dereferenced.
591 (void)new (&_data[_size]) T(*first);
592 ++_size;
593 }
594}

596template<class T, size_t N>
597inline void vec<T, N>::resize(size_type n)
598{
599 n = estd::min(n, N);
600 for (size_type i = n; i < _size; ++i)
601 {
602 _data[i].cast_to_type()->~T();
603 }
604 for (size_type i = _size; i < n; ++i)
605 {
606 (void)new (&_data[i]) T();
607 }
608 _size = n;
609}
