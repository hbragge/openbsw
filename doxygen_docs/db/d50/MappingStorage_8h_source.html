63 // CONFIG_SIZE: number of entries in the mapping config
64 // NUM_STORAGES: number of outgoing storages that will be passed in the constructor
65 // NUM_JOB_SLOTS: number of slots for outgoing jobs
66 template<size_t CONFIG_SIZE, size_t NUM_STORAGES, size_t NUM_JOB_SLOTS>
67 class MappingStorage : public ::storage::MappingStorage
68 {
69  static_assert(CONFIG_SIZE > 0U, "number of blocks must be bigger than 0");
70  static_assert(NUM_STORAGES > 0U, "number of storages must be bigger than 0");
71  static_assert(NUM_JOB_SLOTS > 0U, "number of slots must be bigger than 0");
72 
73 public:
74  // NOTE: the order of "storages" arguments is crucial and must match the indices (outgoingIdx)
75  // given in the config (i.e. the first storage argument corresponds to the outgoingIdx 0,
76  // the second to outgoingIdx 1 and so on)
77  template<typename... T>
78  explicit MappingStorage(
79  MappingConfig const (&config)[CONFIG_SIZE],
80  ::async::ContextType const context,
81  T&&... storages)
82  : ::storage::MappingStorage(
83  reinterpret_cast<MappingConfig const*>(&config),
84  CONFIG_SIZE,
85  context,
86  _inJobs,
87  _outJobPtrs,
88  _storages)
89  , _storages({&storages...})
90  {
91  static_assert(
92  (sizeof...(storages)) == NUM_STORAGES,
93  "number of storage arguments needs to be equal to NUM_STORAGES");
94  _inJobs.fill(nullptr);
95  for (size_t i = 0U; i < NUM_JOB_SLOTS; ++i)
96  {
97  _outJobPtrs[i] = &(_outJobs[i]);
98  }
99  }
100 
101 private:
102  ::etl::array<IStorage* const, NUM_STORAGES> _storages;
103  ::etl::array<StorageJob*, NUM_JOB_SLOTS> _inJobs;
104  ::etl::array<StorageJob, NUM_JOB_SLOTS> _outJobs;
105  ::etl::array<StorageJob*, NUM_JOB_SLOTS> _outJobPtrs;
106 };
