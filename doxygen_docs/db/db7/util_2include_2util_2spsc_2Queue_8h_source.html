43  // The tx.sent and rx.received variables are always in the range [0, 2*N] while the ring
44  // contains N elements. This way there are two distinct constellations where
45  // tx.sent and rx.received point two the same element:
46  // 1) tx.sent == rx.received
47  // 2) tx.sent == rx.received + N
48  // Using this trick the ambiguity between the empty and full cases of the queue is resolved
49  // without the need for an unused element in the queue: Case 1) means "empty" and case 2) means

190 template<class T, size_t N, bool>
191 struct TxData
192 {
193  std::atomic<size_t> sent;
194  std::atomic<size_t> acked;
195  ::etl::array<T, N> data;
196 
197  TxData() : sent(0U), acked(0U), data() {}
198 
199  void reset()
200  {
201  sent.store(0U);
202  acked.store(0U);
203  }
204 
205  template<class RxData, class OnConsumed>
206  void check(RxData const& rxData, OnConsumed const& onConsumed)
207  {
208  size_t numAcked = acked.load();
209  while (numAcked != rxData.received.load())
210  {
211  onConsumed(data[numAcked % N]);
212  numAcked = (numAcked + 1U) % (2U * N);
213  acked.store(numAcked);
214  }
215  }
216 
217  template<class RxData>
218  bool full(RxData const& /*rxData*/) const
219  {
220  size_t const numSent = sent.load();
221  size_t const numAcked = acked.load();
222  return (numSent == ((numAcked + N) % (2U * N)));
223  }
224 };
