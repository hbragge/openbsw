1037template<class T>
1038deque<T>& deque<T>::operator=(deque<T> const& other)
1039{
1040 // same data
1041 if (_data == other._data)
1042 {
1043 return *this;
1044 }
1045
1046 auto const thisSize = static_cast<difference_type>(size());
1047
1048 // verify that we have enough space to copy the elements.
1049 estd_assert(max_size() >= other.size());
1050
1051 if (size() >= other.size())
1052 {
1053 (void)std::copy(other.begin(), other.end(), begin());
1054 (void)erase(cbegin() + static_cast<difference_type>(other.size()), cend());
1055 }
1056 else
1057 {
1058 (void)std::copy(other.begin(), other.begin() + thisSize, begin());
1059 (void)insert(cbegin() + thisSize, other.begin() + thisSize, other.end());
1060 }
1061
1062 return *this;
1063}

1359typename deque<T>::iterator deque<T>::erase(const_iterator const first, const_iterator const last)
1360{
1361 if ((!empty()) && (first != cend()) && (first != last))
1362 {
1363 if (first._current <= _tail)
1364 { // deque payload to be moved is in contiguous memory, only one move required
1365 auto const dst = first._current;
1366 auto const src = last._current;
1367 auto const n = static_cast<size_type>((_tail - src) + 1UL) % _num_elements;
1368 moveLeft(dst, src, n);
1369 _tail
1370 = ((_num_elements + _tail) - static_cast<size_type>(last - first)) % _num_elements;
1371 return iterator(this, first._current);
1372 }
1373 if (last._current > _head)
1374 { // deque payload to be moved is in contiguous memory, only one move required
1375 auto const src = _head;
1376 auto const dst = _head + static_cast<size_type>(last - first);
1377 auto const n = static_cast<size_type>(first._current - src);
1378 moveRight(dst, src, n);
1379 _head = (_num_elements + _head + static_cast<size_type>(last - first)) % _num_elements;
1380 return iterator(this, last._current);
1381 }
1382
1383 // deque payload is wrapped
1384 size_type dst = 0U;
1385 size_type src = last._current;
1386 size_type n = (_tail - src) + 1U;
1387 moveLeft(dst, src, n);
1388 _tail = ((_num_elements + _tail) - (src - dst)) % _num_elements;
1389 src = _head;
1390 dst = static_cast<size_type>(_head + (_num_elements - first._current));
1391 n = first._current - src;
1392 moveRight(dst, src, n);
1393 _head = (_num_elements + _head + (dst - src)) % _num_elements;
1394 return iterator(this, 0U);
1395 }
1396
1397 return end();
1398}

1400template<class T>
1401constructor<T> deque<T>::emplace(const_iterator position)
1402{

1404

1407 return emplace_back();

1409 if (canMoveHeadLeft(position._current))
1410 { // head can be moved left

1414 moveLeft(dst, src, n);

1449template<class T>
1450inline void deque<T>::moveLeft(size_type const destPos, size_type const srcPos, size_type const n)
1451{
1452 if (srcPos == destPos)
1453 {
1454 return; // nothing to do
1455 }
1456 auto const first = iterator(this, srcPos);
1457 auto const last = first + static_cast<difference_type>(n);
1458 auto const d_first = iterator(this, destPos);
1459 auto const d_last = iterator(this, destPos) + static_cast<difference_type>(n);
1460 // would cause undefined behaviour in std::move
1461 estd_assert((d_first < first) || (last <= d_first));
1462
1463 (void)std::move(first, last, d_first);
1464 callDestructors(std::max(d_last, first), last); // destroy leftover elements on the right
1465}

1468template<class T>
1469inline void deque<T>::moveRight(size_type const destPos, size_type const srcPos, size_type const n)
1470{
1471 if (srcPos == destPos)
1472 {
1473 return; // nothing to do
1474 }
1475 auto const first = iterator(this, srcPos);
1476 auto const last = first + static_cast<difference_type>(n);
1477 auto const d_first = iterator(this, destPos);
1478 auto const d_last = iterator(this, destPos) + static_cast<difference_type>(n);
1479 // would cause undefined behaviour in std::move_backward
1480 estd_assert((d_last <= first) || (last < d_last));
1481
1482 (void)std::move_backward(first, last, d_last);
1483 callDestructors(first, std::min(last, d_first)); // destroy leftover elements on the left
