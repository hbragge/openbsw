515template<class T, class Comparator>
516typename ordered_vector<T, Comparator>::const_iterator
517ordered_vector<T, Comparator>::find_or_insert(const_reference value)
518{
519 const_iterator const position = std::lower_bound(cbegin(), cend(), value, _comp);
520 if ((position == cend()) || _comp(value, *position))
521 {
522 if (full())
523 {
524 return end();
525 }
526 (void)_values.insert(position, value);
527 }
528 return position;
529}

552template<class T, class Comparator>
553template<class Predicate>
554void ordered_vector<T, Comparator>::remove_if(Predicate pred)
555{
556 const_iterator b = cbegin();
557 // This is a safe way to remove elements from a container while iterating over the container
558
559 while (b != cend())
560 {
561 if (pred(*b) == true)
562 {
563 b = _values.erase(b);
564 }
565 else
566 {
567 ++b;
568 }
569 }
570}
