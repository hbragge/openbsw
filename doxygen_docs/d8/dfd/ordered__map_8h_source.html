480template<class Key, class T, class Compare>
481inline bool
482operator==(ordered_map<Key, T, Compare> const& lhs, ordered_map<Key, T, Compare> const& rhs)
483{
484 if (lhs.size() != rhs.size())
485 {
486 return false;
487 }
488 for (typename ordered_map<Key, T, Compare>::const_iterator itr = lhs.cbegin();
489 itr != lhs.cend();
490 ++itr)
491 {
492 if (rhs.count(itr->first) > 0U)
493 {
494 if (!(rhs.at(itr->first) == itr->second))
495 {
496 return false;
497 }
498 }
499 else
500 {
501 return false;
502 }
503 }
504 return true;
505}

597template<class Key, class T, class Compare>
598std::pair<typename ordered_map<Key, T, Compare>::iterator, bool>
599ordered_map<Key, T, Compare>::insert(const_reference value)
600{
601 iterator const pos = lower_bound(value.first);
602 if ((pos != end()) && key_eq(pos->first, value.first))
603 {
604 return std::make_pair(pos, false);
605 }
606
607 auto& v = emplace_uninitialized(pos);
608
609 // Constructor needs to be able to modify the memory for placement new.
610 // But v.first is a const object so we cast away the const.
611 (void)constructor<key_type const>(type_utils<key_type const>::const_cast_to_void(&(v.first)))
612 .construct(value.first);
613 (void)constructor<mapped_type>(&v.second).construct(value.second);
614
615 return std::make_pair(pos, true);
616}

666template<class Key, class T, class Compare>
667typename ordered_map<Key, T, Compare>::iterator
668ordered_map<Key, T, Compare>::erase(const_iterator const first, const_iterator const last)
669{
670 estd_assert(last <= cend());
671 if (last <= first)
672 {
673 return begin() + (last - cbegin()); // empty range
674 }
675
676 // copy elements after range to front of range
677 iterator const returnItem = begin() + (first - cbegin());
678 iterator dstIter = returnItem;
679 iterator srcIter = begin() + (last - cbegin());
680 while (srcIter != end())
681 {
682 dstIter->~value_type();
683 (void)constructor<value_type>(type_utils<value_type>::cast_to_raw(dstIter))
684 .construct(::std::move(*srcIter));
685 ++dstIter;
686 ++srcIter;
687 }
688
689 // destroy excess tail elements
690 auto const numToPop = static_cast<size_type>(static_cast<difference_type>(last - first));
691 for (size_type i = 0; i < numToPop; ++i)
692 {
693 _values.pop_back();
694 }
695 return returnItem;
696}

938template<class Key, class T, class Compare>
939typename ordered_map<Key, T, Compare>::size_type
940ordered_map<Key, T, Compare>::lower_bound_search(key_type const& key) const
941{
942 size_type pos;
943 size_type first = 0U;
944 difference_type total = static_cast<difference_type>(_values.size());
945 difference_type step;
946
947 while (total > 0)
948 {
949 pos = first;
950 step = total / 2;
951 pos += static_cast<size_type>(step);
952 if (_comp(_values[pos].first, key) == true)
953 {
954 pos++;
955 first = pos;
956 total -= step + 1;
957 }
958 else
959 {
960 total = step;
961 }
962 }
963 return first;
964}

966template<class Key, class T, class Compare>
967typename ordered_map<Key, T, Compare>::size_type
968ordered_map<Key, T, Compare>::upper_bound_search(key_type const& key) const
969{
970 size_type pos;
971 size_type first = 0U;
972 difference_type total = static_cast<difference_type>(_values.size());
973 difference_type step;
974
975 while (total > 0)
976 {
977 pos = first;
978 step = total / 2;
979 pos += static_cast<size_type>(step);
980 if (_comp(key, _values[pos].first) == false)
981 {
982 pos++;
983 first = pos;
984 total -= step + 1;
985 }
986 else
987 {
988 total = step;
989 }
990 }
991 return first;
992}

994template<class Key, class T, class Compare>
995inline typename ordered_map<Key, T, Compare>::value_type&
996ordered_map<Key, T, Compare>::emplace_uninitialized(ordered_map::iterator const pos)
997{
998 estd_assert(!full());
999 if (pos == cend())
1000 {
1001 return *(_values.emplace_back().release_uninitialized_as());
1002 }
1003
1004 // move all elements after pos to the back by one
1005 auto dstItr = end() - 1;
1006 auto srcItr = dstItr - 1;
1007 _values.emplace_back().construct(::std::move(*dstItr)); // construct new last element
1008 dstItr->~value_type(); // destroy previous last element
1009 while (dstItr != pos)
1010 {
1011 // move src to dest and destroy src
1012 (void)constructor<ordered_map::value_type>(
1013 type_utils<ordered_map::value_type>::cast_to_raw(dstItr))
1014 .construct(::std::move(*srcItr));
1015 srcItr->~value_type();
1016 --dstItr;
1017 --srcItr;
1018 }
1019 return *pos;
1020}
